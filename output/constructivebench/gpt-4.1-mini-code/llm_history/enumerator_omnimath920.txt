```python
# Enumerate 3-digit Armstrong numbers (n = sum of cubes of its digits)

def digits(n):
    return [int(d) for d in str(n)]

print("Checking 3-digit numbers for Armstrong property:")
for n in range(100, 1000):
    d = digits(n)
    s = sum(x**3 for x in d)
    if s == n:
        print(n)
```
