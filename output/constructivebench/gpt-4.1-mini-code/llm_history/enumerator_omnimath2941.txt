The problem is to count triples of positive integers (a,b,c) with 6*a*b = c^2, a < b < c â‰¤ 35. We will enumerate all such triples and count them, printing each when found.

```python
# Enumerate positive triples (a,b,c) with 6*a*b = c^2, a<b<c<=35
# Print each triple found and total count at end.

def main():
    limit = 35
    count = 0
    for c in range(1, limit+1):
        c2 = c*c
        # Since 6 * a * b = c^2, c^2 must be divisible by 6*a
        # Try a from 1 to c-2 (since a<b<c)
        for a in range(1, c-1):
            if (6 * a) == 0:
                continue
            if c2 % (6 * a) != 0:
                continue
            b = c2 // (6 * a)
            if a < b < c and b > 0:
                print(f"(a,b,c)=({a},{b},{c}) satisfies 6*a*b=c^2")
                count += 1
    print(f"Total count of triples (a,b,c): {count}")

if __name__ == "__main__":
    main()
```
