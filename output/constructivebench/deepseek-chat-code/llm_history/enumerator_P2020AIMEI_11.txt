This problem involves finding all integer triples (a, b, c) where |a|, |b|, |c| â‰¤ 10, and there exists a d such that the composition g(f(x)) has roots at x=2 and x=4. We'll enumerate all possible (a, b, c) combinations and check the conditions.

```python
def main():
    count = 0
    for a in range(-10, 11):
        for b in range(-10, 11):
            for c in range(-10, 11):
                # Calculate f(2) and f(4)
                f2 = 4 + 2*a + b
                f4 = 16 + 4*a + b
                
                # g(f2) = f2^2 + c*f2 + d = 0
                # g(f4) = f4^2 + c*f4 + d = 0
                # Subtract to eliminate d:
                # (f2^2 - f4^2) + c*(f2 - f4) = 0
                # (f2 - f4)(f2 + f4 + c) = 0
                
                # Case 1: f2 = f4
                if f2 == f4:
                    # Then any d = -f2^2 - c*f2 works
                    count += 1
                    print(f"Found solution: a={a}, b={b}, c={c} (case f2=f4)")
                else:
                    # Case 2: f2 + f4 + c = 0
                    if f2 + f4 + c == 0:
                        # Then d = -f2^2 - c*f2
                        count += 1
                        print(f"Found solution: a={a}, b={b}, c={c} (case f2+f4+c=0)")
    
    print(f"Total number of solutions: {count}")

if __name__ == "__main__":
    main()
```
