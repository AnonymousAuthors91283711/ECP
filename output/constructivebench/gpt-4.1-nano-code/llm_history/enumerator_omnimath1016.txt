import math

def totient(n):
    """Compute Euler's totient function of n."""
    result = n
    p = 2
    while p * p <= n:
        if n % p == 0:
            while n % p == 0:
                n //= p
            result -= result // p
        p += 1 if p == 2 else 2
    if n > 1:
        result -= result // n
    return result

def order_mod(n, m):
    """Compute the order of m modulo n, assuming gcd(m, n) == 1."""
    if math.gcd(m, n) != 1:
        return None
    result = 1
    x = m % n
    while x != 1:
        x = (x * m) % n
        result += 1
        if result > n:  # Just a safeguard; order can't be more than n
            return None
    return result

max_n = 100
threshold = 10  # to limit computations

for n in range(1, max_n + 1):
    phi_n = totient(n)
    condition_holds = True
    for m in range(1, n):
        if math.gcd(m, n) == 1:
            ord_m = order_mod(n, m)
            if ord_m is None:
                continue
            if ord_m >= phi_n / 10:
                condition_holds = False
                break
    if condition_holds:
        print(f"Candidate n = {n}")
